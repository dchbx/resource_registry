def call(input)
  if defined? ResourceRegistry::AppSettings
    DeepMerge.deep_merge!(
      input.to_h,
      ResourceRegistry::AppSettings,
      merge_hash_arrays: true,
      merge_nil_values:  true,
    )
  else
    ResourceRegistry.const_set('AppSettings', input.to_h)
  end

  Success(input)
end
