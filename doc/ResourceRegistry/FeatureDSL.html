def setting(id)
  if !id.is_a?(String) && !id.is_a?(Symbol)
    raise ArgumentError, "#{id} must be a String or Symbol"
  end

  id = id.to_sym
  @feature.settings.detect { |setting| setting.key == id }
end

def settings(id = nil)
  return [] unless @feature.settings
  id == nil ? @feature.settings : setting(id)
end
